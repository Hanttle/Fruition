local function TTAG_fake_script() -- CreditTextButton.creditopen 
	local script = Instance.new('LocalScript', CreditTextButton)

	script.Parent.MouseButton1Click:Connect(function()
		script.Parent.Parent.Parent.GameScrollingFrame.Visible = false
		script.Parent.Parent.Parent.CreditFrame.Visible = true
		script.Parent.Parent.Parent.ScriptScrollingFrame = false
	end)
	
end
